import pymysql
HOST="localhost"
PORT=3307
USER = "webdata"
PASSWORD = "123456"
DATABASE = "Webdatabase"
#获取数据库连接
conn=pymysql.connect(host=HOST,port=PORT,user=USER,password=PASSWORD,db=DATABASE)
#获取数据库游标
cur=conn.cursor()
# 创建sql语句
sql = "insert into coment_info(title,url,author,time) values ('%s','%s','%s','%s')" % (
"职业大赛2","http://www.gzeic.com","yangyu","2022-11-25"
)
# 执行sql语句
cur.execute(sql)
# 提交 不要数据插入和查询操作也能成功
#conn.commit()

#读取数据打印
cur.execute("select * from coment_info")
results=cur.fetchall()
for row in results:
    print(row)

#关闭数据库
#关闭游标对象
cur.close()
conn.close()
